	\item	$\ds y=f(x)=x/(1-x)$
	\item	$\ds y=f(x)=-\sqrt{-x}$

\begin{ex}
Suppose $f(x) = 3x-9$ and $\ds g(x) = \sqrt{x}$.  What is the
domain of the composition $(g\circ f)(x)$?
\begin{sol}
$\ds \{x\mid x\ge3\}$, $\{x\mid x\ge0\}$
\end{sol}
